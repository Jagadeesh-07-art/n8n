    <script>
        function updateCountdown() {
            const now = new Date();
            const targetDate = new Date('December 31, 2026');
            
            // Calculate the difference in days
            const timeDifference = targetDate - now;
            const daysRemaining = Math.ceil(timeDifference / (1000 * 60 * 60 * 24));
            
            // Update the countdown display
            document.getElementById('countdown').textContent = daysRemaining;
        }

        // Update the countdown immediately and then every day
        updateCountdown();
        setInterval(updateCountdown, 86400000); // Update every 24 hours
    </script>
